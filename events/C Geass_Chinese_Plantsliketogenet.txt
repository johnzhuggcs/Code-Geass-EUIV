# Crown Prince and Eunuch
country_event = {
	id = chinese_han.10
	title = "unhappyPrince.1.t"
	desc = "unhappyPrince.1.d"
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = CHF
			tag = HDS
		}
		AND = {
			is_year = 2005
			heir_age = 20
			heir_has_ruler_dynasty = yes
			NOT = {
				faction_in_power = enuchs
			}
			NOT = {
				has_country_flag = unhappy_prince_killed
			}
			NOT = {
				has_country_flag = enuchs_supported_coup
			}
		}
		
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "unhappyPrince.1.a"
		kill_ruler = yes
		add_heir_personality = cruel_personality
		add_heir_personality = intricate_web_weaver_personality
		add_heir_personality = infertile_personality
		add_faction_influence = {
			faction = enuchs
			influence = 100
		}
		set_country_flag = enuchs_supported_coup
		country_event = {
			id = chinese_han.12
			days = 0
			tooltip = "Long Live the Empire!"
		}
	}
	
	option = {
		name = "unhappyPrince.1.b"
		exile_heir_as = "Imprisoned"
		country_event = {
			id = chinese_han.11
			days = 3
			tooltip = "Alas, my son!"
		}
	}
		
}

# Jiang Lihua/Tianzi

	id = chinese_han.13
	title = "tianzi.1.t"
	desc = "tianzi.1.d"
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = CHF
			tag = HDS
		}
		AND = {
			is_year = 2008
			has_country_flag = enuchs_supported_coup
			faction_in_power = enuchs
			ruler_age = 23
			NOT = {
				has_country_flag = unhappy_prince_killed
			}
			
		}
		
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = "tianzi.1.a"

			name = "Lihua"
			female = yes
			dynasty = ROOT
			age = 2
			claim = 30
			adm = 6
			dip = 3
			mil = 0
		}
		add_ruler_personality = naive_personality
		add_ruler_personality = benevolent_personality
		add_stability = -3
		add_corruption = 60
		set_country_flag = enuchs_further_power
	}
	
	option = {
		name = "tianzi.1.b"
		define_ruler = {
			name = "Shuolan"
			dynasty = ROOT 
			age = 23
			regency = no
			female = no
			adm = 5
			dip = 5
			mil = 6
		}
	}
}
